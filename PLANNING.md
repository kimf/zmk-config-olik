## PLACED

-_ = ovanför höger pil
'" = UNDER X

page up/down = på pilarna på sym
home/end = på pilarna på sym

caps = på B på media

°/§ = på num "extra keys" då den inte används så ofta
£   = på sym extra keys då den inte används så ofta

@  = på B sym layer, dubbel-tap = "@gmail.com"
~  = på "N" sym layer, dubbel-tap = "~/"
`´ = på P på sym layer
^  = på V på sym layer
# = på Ö på sym och num
? = på Ä på sym och nym
! = Y på sym
& = U på sym
$ = I på sym
/\ = O på sym
|  = O på num (alt + O på sym borde också funka)

< = PÅ A på num, dubbel-tap på = </> + vänster
> = PÅ S på num, dubbel-tap = />

( = PÅ H på sym - dubbel-tap stänger och flyttar vänster
[ = på J på sym - dubbel-tap stänger och flyttar vänster
{ = på K på sym - dubbel-tap stänger och flyttar vänster

) = på G på sym
] = på F på sym
} = på D på sym

+ = L på sym och num
= = på M på sym och num (≈ borde komma automatiskt med alt + =)
% = på N på num layer
* = K på num
